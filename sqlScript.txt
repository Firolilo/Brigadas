CREATE TABLE brigada (
  id INT IDENTITY PRIMARY KEY,
  nombre VARCHAR(120) NOT NULL,
  cantidadactivos INT NOT NULL DEFAULT 0,
  nombrecomandante VARCHAR(120) NOT NULL,
  celularcomandante VARCHAR(30) NULL,
  encargadologistica VARCHAR(120) NULL,
  celularlogistica VARCHAR(30) NULL,
  numerosemergencia VARCHAR(30) NULL
);

CREATE TABLE epp_ropa (
  id INT IDENTITY PRIMARY KEY,
  brigadaid INT NOT NULL,
  item VARCHAR(80) NOT NULL,
  xs INT NOT NULL DEFAULT 0,
  s INT NOT NULL DEFAULT 0,
  m INT NOT NULL DEFAULT 0,
  l INT NOT NULL DEFAULT 0,
  xl INT NOT NULL DEFAULT 0,
  observaciones VARCHAR(400) NULL,
  FOREIGN KEY (brigadaid) REFERENCES brigada(id)
);

CREATE TABLE botas (

  brigadaid INT NOT NULL,

  talla42 INT NOT NULL DEFAULT 0,

);
